tasks.register("printAllDependencies") {
    group = "help"
    description = "Dump all .jar files from every non-test RuntimeClasspath"
    doLast {
        // find every configuration whose name ends with "RuntimeClasspath"
        // but exclude unit-test & android-test variants
        def runtimeConfs = configurations.findAll { conf ->
            conf.name.endsWith("RuntimeClasspath") &&
                    !conf.name.toLowerCase().contains("unittest") &&
                    !conf.name.toLowerCase().contains("androidtest")
        }

        runtimeConfs.each { conf ->
            println "\n--- ${conf.name} ---"
            try {
                // resolve() will give us all files for that configuration
                conf.resolve()
                        .findAll { it.name.endsWith(".jar") }
                        .each { jar ->
                            println jar.absolutePath
                        }
            } catch (Exception e) {
                println "⚠️  failed to resolve ${conf.name}: ${e.message}"
            }
        }
    }
}
